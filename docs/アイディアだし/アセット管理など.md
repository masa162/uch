Cloudflare R2 バケットの作成

Cloudflareにログインし、「R2」メニューに移動します。

「バケットを作成する」をクリックし、バケット名（例: uchinokiroku-assets）を決定します。全世界でユニークな名前である必要はありません。

ロケーションは「自動」のままでOKです。

R2の概要ページ右上にある「R2 API トークンを管理」をクリックします。

「APIトークンを作成」をクリックし、トークン名を付けます（例: uchinokiroku-vps-worker）。

権限は「管理者読み取り&書き込み」を選択して作成します。

重要: 作成後に表示される「アクセスキーID」と「シークレットアクセスキー」を必ずコピーして、安全な場所に保管してください。この画面を閉じると二度と表示されません。

## 2. 設計方針：安全で高速な「署名付きURL」方式の採用
次に、どうやってファイルをアップロードするかを決めます。最も推奨されるのが「署名付きURL（Presigned URL）」方式です。

この方式のメリットと流れ
ユーザーのブラウザからVPS経由で巨大なファイルをR2に送ると、VPSに大きな負荷がかかり、時間もかかります。そこで、以下の流れでVPSの負荷をなくし、高速かつ安全にアップロードします。

ブラウザ → Next.jsサーバー: 「こういうファイルをアップロードしたいんだけど、専用の許可証（URL）をちょうだい」とリクエスト。

Next.jsサーバー ↔ Cloudflare R2: サーバーだけが知っている秘密鍵を使って、R2に「このファイル名で、10分間だけアップロードを許可する特別なURLを発行して」と依頼。

Next.jsサーバー → ブラウザ: R2から受け取った「署名付きURL」をブラウザに渡す。

ブラウザ → Cloudflare R2: ブラウザは受け取ったURLめがけて、ファイルのデータを直接送信（アップロード）する。

この方法なら、ファイル本体はVPSサーバーを経由しないため、サーバーは非常に身軽なままです。

pages/api/r2/generate-upload-url.js

JavaScript

import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";
import { getSignedUrl } from "@aws-sdk/s3-request-presigner";
import { randomUUID } from 'crypto';

// S3クライアントをR2設定で初期化
const s3Client = new S3Client({
  region: "auto",
  endpoint: process.env.R2_S3_API_ENDPOINT,
  credentials: {
    accessKeyId: process.env.CLOUDFLARE_ACCESS_KEY_ID,
    secretAccessKey: process.env.CLOUDFLARE_SECRET_ACCESS_KEY,
  },
});

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ message: 'Method not allowed' });
  }

  try {
    const { fileName, fileType } = req.body;

    if (!fileName || !fileType) {
      return res.status(400).json({ message: 'Missing fileName or fileType' });
    }
    
    // ファイル名が重複しないようにユニークなIDを付与
    const uniqueFileName = `${randomUUID()}-${fileName}`;

    // 署名付きURLを生成するためのコマンドを作成
    const command = new PutObjectCommand({
      Bucket: process.env.R2_BUCKET_NAME,
      Key: uniqueFileName,
      ContentType: fileType,
    });

    // 署名付きURLを生成（有効期限は3600秒 = 1時間）
    const signedUrl = await getSignedUrl(s3Client, command, { expiresIn: 3600 });

    // 生成したURLとファイル名をクライアントに返す
    res.status(200).json({
      url: signedUrl,
      fileName: uniqueFileName,
    });

  } catch (error) {
    console.error('Error creating signed URL:', error);
    res.status(500).json({ message: 'Error creating signed URL' });
  }
}

components/ImageUploader.js

JavaScript

import { useState } from 'react';

export default function ImageUploader() {
  const [selectedFile, setSelectedFile] = useState(null);
  const [status, setStatus] = useState('initial');

  const handleFileChange = (e) => {
    if (e.target.files) {
      setSelectedFile(e.target.files[0]);
      setStatus('initial');
    }
  };

  const handleUpload = async () => {
    if (!selectedFile) return;

    setStatus('uploading');

    try {
      // 1. 署名付きURLをリクエスト
      const response = await fetch('/api/r2/generate-upload-url', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          fileName: selectedFile.name,
          fileType: selectedFile.type,
        }),
      });

      if (!response.ok) throw new Error('Failed to get signed URL.');

      const { url, fileName } = await response.json();
      console.log('Got signed URL:', url);

      // 2. 署名付きURLにファイルを直接PUTリクエストでアップロード
      const uploadResponse = await fetch(url, {
        method: 'PUT',
        body: selectedFile,
        headers: { 'Content-Type': selectedFile.type },
      });

      if (!uploadResponse.ok) throw new Error('Upload failed.');
      
      setStatus('success');
      console.log('Upload successful! File available at:', fileName);
      // ここで、fileNameを親コンポーネントに渡したり、
      // DBに保存する処理をトリガーするAPIを呼んだりする

    } catch (error) {
      console.error(error);
      setStatus('fail');
    }
  };

  return (
    <div>
      <input type="file" accept="image/*" onChange={handleFileChange} />
      <button onClick={handleUpload} disabled={!selectedFile || status === 'uploading'}>
        {status === 'uploading' ? 'アップロード中...' : 'アップロード'}
      </button>

      {status === 'success' && <p>✅ アップロードが完了しました！</p>}
      {status === 'fail' && <p>❌ アップロードに失敗しました。</p>}
    </div>
  );
}
